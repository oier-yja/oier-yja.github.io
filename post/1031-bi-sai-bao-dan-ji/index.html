T1互不攻击的炮

光看题面很简单的一道裸爆搜，n≤6n\le6n≤6直接dfsdfsdfs暴力枚举当前格子的放与不放，每放一个ans++ans++ans++。
但本蒟蒻不满足于此
要是nnn很大时dfsdfsdfs肯定炸，而在当前的格子选择放与不放炮，显然跟前面放与不放炮的方案数有着关系。
首先需要一个状态记录当前正在处理的行列，根据中国象棋规则，在同一行和同一列中肯定是不能放≥2\ge2≥2个炮的 。所以只需要一位iii表示已经处理到第iii行。而当前行的放与不放问题，分为放一个还是放两个。而要在这一列中放一个，那么所放的那一列肯定不能大于1个棋子。所以要另外两个状态，有jjj列放1个棋子，有kkk列放2个棋子。

设dp[i][j][k]dp[i][j][k]dp[i][j][k]为当前第iii行有jjj列放一个棋子，有kkk列放两个棋子所能得到的方案数。

综上所述可简化如下---
在第iii行上

  • 不放
    • 直接继承上一行的方案数
      • dp[i][j][k]=dp[i−1][j][k]dp[i][j][k]=dp[i-1][j][k]dp[i][j][k]=dp[i−1][j][k]
  • 放一个
    • 在已放了一个棋子的那一列上放
      • 当前有jjj列放了一个，而因为在上一状态放了一个棋子列中放下了一个棋子，所以会使上一状态放了一个棋子列数量减1，而放了两个棋子列中，会因放在已经放了一个棋子那一列中再放一个棋子变成已经放了两个棋子的列，所以放两个棋子列数要加1。
      • dp[i][j][k]+=dp[i−1][j+1][k−1]∗(j+1)dp[i][j][k]+=dp[i-1][j+1][k-1]*(j+1)dp[i][j][k]+=dp[i−1][j+1][k−1]∗(j+1)
    • 在还未放棋子的那一列上放
      • 当前有jjj列放了一个kkk列放了两个，而当前状态放是由上一状态转移来的，因为在未放棋子那一列上放棋子，所以放了两个棋子的列数kkk不变，而放了一个棋子的列数jjj将会增加一。换言之如果当前有jjj列放一个，则上一状态有j−1j-1j−1列放一个。
      • dp[i][j][k]+=dp[i−1][j−1][k]∗(m−(j−1)−k)dp[i][j][k]+=dp[i-1][j-1][k]*(m-(j-1)-k)dp[i][j][k]+=dp[i−1][j−1][k]∗(m−(j−1)−k)
  • 放两个
    • 在已经放了一个棋子那一列上放一个，在还未放棋子的那一列上放一个
      • 在已放了一个棋子列上放下一个会使得jjj减一而kkk加一，再在未放棋子的那一列上放jjj又加一，故jjj不变kkk加一。换言之上一个状态为dp[i−1][j][k−1]dp[i-1][j][k-1]dp[i−1][j][k−1]。
      • dp[i][j][k]+=dp[i−1][j][k−1]∗j∗(m−(j−1)−k)dp[i][j][k]+=dp[i-1][j][k-1]*j*(m-(j-1)-k)dp[i][j][k]+=dp[i−1][j][k−1]∗j∗(m−(j−1)−k)
    • 在两列都未放棋子的列上放两个
      • 在两个都未放棋子列上放两个棋子，故j+2j+2j+2，kkk不变。反之则上一状态为dp[i−1][j−2][k]dp[i-1][j-2][k]dp[i−1][j−2][k]。在m−(j−2)−km-(j-2)-km−(j−2)−k中选两个，用组合数学公式C(m−(j−2)−k,2)C(m-(j-2)-k,2)C(m−(j−2)−k,2)求出方案数。
      • dp[i][j][k]+=dp[i−1][j−2][k]∗C(m−(j−2)−k)dp[i][j][k]+=dp[i-1][j-2][k]*C(m-(j-2)-k)dp[i][j][k]+=dp[i−1][j−2][k]∗C(m−(j−2)−k)
    • 在两列都放了一个棋子上放下两个棋子
      • jjj会因为多放两个棋子后变为放两个棋子的列，故减二，kkk也因此加二，反之则上一状态为 dp[i−1][j+2][k−2]dp[i-1][j+2][k-2]dp[i−1][j+2][k−2]。
      • dp[i][j][k]+=dp[i−1][j+2][k−2]∗C(j+2,2)dp[i][j][k]+=dp[i-1][j+2][k-2]*C(j+2,2)dp[i][j][k]+=dp[i−1][j+2][k−2]∗C(j+2,2)

综上得出代码

#include<bits/stdc++.h>
using namespace std;
const int N=105,M=105;
inline long long read(){
	long long x=0,w=0;char ch=0;
	while(!isdigit(ch)){w|=ch=='-',ch=getchar();}
	while(isdigit(ch))x=(x<<3)+(x<<1)+(ch^48),ch=getchar();
	return w?-x:x;
}
inline void write(long long x){
	if(x<0)putchar('-'),x-=x;
	if(x>9)write(x/10);
	putchar(x%10+'0');
	return ;
}
long long n,m,mod=9999973,dp[N][M][M],ans=0;
long long C(long long x){
	return (x*(x-1)/2)%mod;
}
int main(){
	n=read();
	m=read();
	dp[0][0][0]=1;
	for(register int i=1;i<=n;i++){
		for(register int j=0;j<=m;j++){
			for(register int k=0;k<=m;k++){
				dp[i][j][k]=dp[i-1][j][k];
				if(k>=1)//放有一个上 
					dp[i][j][k]+=dp[i-1][j+1][k-1]*(j+1);
				if(j>=1)//放没有一个上 
					dp[i][j][k]+=dp[i-1][j-1][k]*(m-(j-1)-k);
					
				if(k>=1)//放一个有，一个没有上 
					dp[i][j][k]+=dp[i-1][j][k-1]*j*(m-j-(k-1));
				if(j>=2)//放两个没有 
					dp[i][j][k]+=dp[i-1][j-2][k]*C(m-(j-2)-k);
				if(k>=2)//放两个有 
					dp[i][j][k]+=dp[i-1][j+2][k-2]*C(j+2);
				dp[i][j][k]%=mod;
			}
		}
	}
	for(register int i=0;i<=m;i++)
		for(register int j=0;j<=m;j++)
			ans+=dp[n][i][j];
	write(ans);
	return 0;
}

进行预处理，开一个结构体aaa包括编号iii，品种kkk。将以iii为关键字从小到大排序方便处理。
对于拍照截取的奶牛列，设为′G′'G'′G′种的为1，反之为-1，求其前缀和sum[i]sum[i]sum[i]，可以便迅求出[i,j][i,j][i,j]中′G′'G'′G′种奶牛的个数。。
分为两种情况进行讨论：

  • 全为一个品种
    • 用尺取法选取最大连续同一种类长度
  • 为不同品种
    • 要满足所选区间[i,j][i,j][i,j]中奶牛数量得相同所以sum[i+1]−sum[j]=0sum[i+1]-sum[j]=0sum[i+1]−sum[j]=0：因为′G′'G'′G′种为1′H′'H'′H′种为-1，如果[i,j][i,j][i,j]区间中之和为0，明显′G′'G'′G′种和′H′'H'′H′种数量是一样的，这样加起来才为0而不是>0>0>0或<0<0<0。对于每个sum[i]sum[i]sum[i]记录第一次出现的位置和最后一次出现的位置，其中间肯定是种类数复合条件的，就求其长度记录。
#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
inline int read(){
	int x=0,w=0;char ch=0;
	while(!isdigit(ch)){w|=ch=='-',ch=getchar();}
	while(isdigit(ch))x=(x<<3)+(x<<1)+(ch^48),ch=getchar();
	return w?-x:x;
}
inline void write(int x){
	if(x<0)putchar('-'),x-=x;
	if(x>9)write(x/10);
	putchar(x%10+'0');
	return ;
}
struct node{
	int k;
	int b;
	bool operator < (const node &a) const {
		return b<a.b;
	}
}a[N];
int n,sum[N],ans,vis[N];
int main(){
	n=read();
	for(register int i=1;i<=n;i++){
		a[i].b=read();
		char k;
		cin>>k;	
		a[i].k=(k=='G')?1:-1;
	}
	sort(a+1,a+1+n);
	sum[0]=n+1;
	sum[1]=sum[0]+a[1].k;
	int l=1;
	for(register int i=2;i<=n;i++){
		if(a[i].k!=a[i-1].k)
			l=i;
		ans=max(ans,a[i].b-a[l].b);
		sum[i]=sum[i-1]+a[i].k;
	}
	for(register int i=1;i<=n;i++){
		if(!vis[sum[i]]){
			vis[sum[i]]=i+1;
		}else{
			ans=max(ans,a[i].b-a[vis[sum[i]]].b);
		}
	}
	write(ans);
	return 0;
}

使用优先队列存地铁票，以时间为第一关键字，价格为第二关键字，越早的地铁票越优先被使用，价格越便宜的地铁票越优先处理。而每当有新的乘车记录时，将队列中所有tq−tn>=45t_q-t_n>=45tq​−tn​>=45的全pop掉，来维护队列中所有地铁票都是没过期的。如果当前处理的票是地铁票直接放队列，反之从队列中找出第一个大于等于车票的地铁票进行使用。

#include<bits/stdc++.h>
using namespace std;
const int N=100005;
inline int read(){
	int x=0,w=0;char ch=0;
	while(!isdigit(ch)){w|=ch=='-',ch=getchar();}
	while(isdigit(ch))x=(x<<3)+(x<<1)+(ch^48),ch=getchar();
	return w?-x:x;
}
inline void write(int x){
	if(x<0)putchar('-'),x-=x;
	if(x>9)write(x/10);
	putchar(x%10+'0');
	return ;
}
struct node{
	int w;
	int t;
	node(int w_,int t_){
		w=w_;
		t=t_;
	}
	bool operator < (const node &a) const {
		if(a.t!=t)
			return a.t<t;
		return a.w<w;
	}
};
priority_queue<node> q;
queue<node> v;
int n,ans;
int main(){
	n=read();
	for(register int i=1;i<=n;i++){
		int c=read(),w=read(),t=read();
		ans+=w;
		if(c==0){
			q.push(node(w,t));
		}else{
			while(!q.empty() && t-q.top().t>45)
				q.pop();
			
			while(!q.empty() && q.top().w<w){
				v.push(q.top());
				q.pop();
			}
			if(!q.empty()){
				ans-=w;
				q.pop();
			}
			while(!v.empty()){
				q.push(v.front());
				v.pop();
			}
		}
	}
	write(ans);
	return 0;
}

T2对称二叉树

没什么技巧可言，直接上dfs暴力判断以每个节点为根的二叉树是否为对称二叉树，再用ans比较出max的。
问题在于对二叉树的判定，对于以iii为根的二叉树要满足:

  • 其左右儿子权值相等或都为空
  • 其左儿子的左儿子与右儿子的右儿子满足上述条件(这样才能是一棵对称二叉树嘛)
  • 其左儿子的右儿子与右儿子的左儿子满足上述条件

以此类推，直到当前iii为空，如果满足条件每次就返回truetruetrue并顺便统计下节点个数(cnt+=2)最后加上根节点本身，至此判定出是否为对称二叉树并求出节点数

#include<bits/stdc++.h>
using namespace std;
const int N=1000005;
inline int read(){
	int x=0,w=0;char ch=0;
	while(!isdigit(ch)){w|=ch=='-',ch=getchar();}
	while(isdigit(ch))x=(x<<3)+(x<<1)+(ch^48),ch=getchar();
	return w?-x:x;
}
inline void write(int x){
	if(x<0)putchar('-'),x-=x;
	if(x>9)write(x/10);
	putchar(x%10+'0');
	return ;
}
int n,w[N],cnt,ans,l[N],r[N];
bool flag; 
bool dfs(int ls,int rs){
	if(flag)
		return false;
	if(ls!=-1){
		cnt++;
	}
	if(rs!=-1){
		cnt++;
	}
	if(ls==-1 && rs==-1)
		return true;
	if((ls==-1 || rs==-1) || (w[ls]!=w[rs])){
		flag=1;
	 	return false;
	} 
	if(!dfs(l[ls],r[rs]) || !dfs(r[ls],l[rs])){
		flag=1;
		return false;
	}
	return true;
}
int main(){
	n=read();
	for(register int i=1;i<=n;i++)
		w[i]=read();
	for(register int i=1;i<=n;i++)
		l[i]=read(),r[i]=read();
	
	for(register int i=1;i<=n;i++){
		flag=0;
		cnt=1;
		if(dfs(l[i],r[i]))
			ans=max(ans,cnt);
	}
	write(ans);
	return 0; 
}
